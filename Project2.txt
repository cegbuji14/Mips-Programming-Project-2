.data
    userInput: .space 1000

.text
    main:
    #Lets users input a 1000 character string
      li $v0, 8
      la $a0, userInput
      li $a1, 1001
      syscall
      
    start:
    #Loads the address of inputted string into register $t1, puts first bit (character) of string in $s0
      la $t1, userInput
      add $t1, $t1, $t0
      lb $s0, ($t1)
      beq $t0, 1000, conditions #if it has reached the 1000th character, start looping through the remaining characters
      
    conditions:
